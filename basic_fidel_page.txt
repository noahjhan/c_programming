    <script>
        // JavaScript to change the background color based on scroll position
        const colors = [
            // "#f0e6d2",  // Cream color
            "#458588", // (69, 133, 136)
            "#d79921", // (215, 153, 33)
            "#cc241d", // (204, 36, 29)
            "#3c3836", // (60, 56, 54)
        ];

        window.addEventListener('scroll', () => {
            const scrollPercentage = document.documentElement.scrollTop / (document.documentElement.scrollHeight - window.innerHeight);
            const colorIndex = Math.floor(scrollPercentage * (colors.length - 1));
            document.body.style.backgroundColor = colors[colorIndex];
        });
    </script>
